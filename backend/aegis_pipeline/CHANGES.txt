def poll_status(pdf_id, headers, poll_interval=30, max_polls=30):
    url = f"https://api.mathpix.com/v3/pdf/{pdf_id}.json"
    last_completed = 0
    stuck_count = 0
    
    for poll_count in range(max_polls):
        print(f"Polling attempt {poll_count + 1} for PDF ID {pdf_id}")
        response = requests.get(url, headers=headers)
        status_data = response.json()
        print("Polling Status:", status_data)
        
        if status_data.get("status") == "completed":
            return status_data
            
        # Check if processing is stuck
        current_completed = status_data.get("num_pages_completed", 0)
        if current_completed == last_completed:
            stuck_count += 1
            if stuck_count > 5:  # If no progress for 5 polls
                print(f"Processing stuck at page {current_completed}")
                return None
        else:
            stuck_count = 0
            last_completed = current_completed
            
        time.sleep(poll_interval)
    return None

def process_with_mathpix(file):
    options = {
        "conversion_formats": {"docx": True, "tex.zip": True},
        "math_inline_delimiters": ["$", "$"],
        "rm_spaces": True
    }

    r = requests.post("https://api.mathpix.com/v3/pdf",
        headers={
            "app_id": MATHPIX_APP_ID,
            "app_key": MATHPIX_API_KEY
        },
        data={
            "options_json": json.dumps(options)
        },
        files={
            "file": (file.filename, file.stream, file.content_type)
        }
    )

    API_resp = r.json()
    pdf_id = API_resp.get("pdf_id")
    if not pdf_id:
        return None

    headers = {
        "app_key": MATHPIX_API_KEY,
        "app_id": MATHPIX_APP_ID
    }

    status_data = poll_status(pdf_id, headers)
    if not status_data:
        return None

    url = f"https://api.mathpix.com/v3/pdf/{pdf_id}.mmd"
    response = requests.get(url, headers=headers)
    mmd_content = response.text

    mmd_content = convert_to_katex_format(mmd_content)
    temp_file = tempfile.NamedTemporaryFile(delete=False, suffix=".txt")
    temp_file.write(mmd_content.encode('utf-8'))
    temp_file.close()

    return temp_file.name

def clean_question_text(q: str) -> str:
    if not q:
        return q
    q = q.strip()
    # Remove leading question numbers like "1.", "39.", "Q1.", "Q.49"
    q = re.sub(r'^\s*(?:Q\.?\s*\d+\.?|\d+\.?)\s*', '', q)
    # Remove trailing "..... [4]" or "[3]" style marks (any dots + bracketed number at end)
    q = re.sub(r'\.*\s*\[\s*\d+\s*\]\s*$', '', q)
    return q.strip()def convert_to_katex_format(latex_text):
    if not latex_text:
        return latex_text
    
    latex_text = latex_text.replace("\section*", "")
    latex_text = latex_text.replace("$\qquad$", "__")
    latex_text = latex_text.replace(r'\begin{tabular}', r'\begin{array}')
    latex_text = latex_text.replace(r'\end{tabular}', r'\end{array}')
    latex_text = re.sub(r'\\begin{array}{\|[lcr|]+\|}', r'\\begin{array}{|l|l|l|l|}', latex_text)
    latex_text = latex_text.replace('$', '')
    latex_text = re.sub(r'\\(?:mathrm|mathbf)\{([^\}]+)\}', r'\1', latex_text)
    
    def wrap_text(match):
        return r'\text{' + match.group(1).strip() + '}'
    latex_text = re.sub(r'(?<=\|)([^&\\]+)(?=\s*&)', wrap_text, latex_text)

    latex_text = re.sub(r'\bfi\b', r'f_i', latex_text)
    latex_text = re.sub(r'\bxi\b', r'x_i', latex_text)
    latex_text = re.sub(r'\bxif\b', r'x_i f_i', latex_text)
    latex_text = re.sub(r'(\d+\.?\d*)\s+f\b', r'\1f', latex_text)
    latex_text = re.sub(r'(\\end{array})', r'\1\\[1em]', latex_text)
    latex_text = re.sub(r'(\\text{Mean})', r'\n\1', latex_text)
    latex_text = re.sub(r'^\s*(\d+)\.', r'\\textbf{\1.}\\\\[1em]', latex_text)
    latex_text = re.sub(r'\\underline{([^}]+)}\s*=\s*(\d+\.?\d*)', r'\\underline{\1} = \2', latex_text)
    
    return latex_text.strip()

def parse_questions(file_path):
    with open(file_path, 'r', encoding='utf-8') as file:
        content = file.read()
    
    question_blocks = re.split(r'(\n\d+[).])', content)
    questions = []
    current_question = ""
    
    for block in question_blocks:
        if re.match(r'^\d+[).]', block.strip()):
            if current_question:
                match = re.match(r'^(\d+)[).]\s*(.*)', current_question.strip(), re.DOTALL)
                if match:
                    questions.append({
                        'number': int(match.group(1)),
                        'text': current_question.strip()
                    })
            current_question = block
        else:
            current_question += block
    
    if current_question:
        match = re.match(r'^(\d+)[).]\s*(.*)', current_question.strip(), re.DOTALL)
        if match:
            questions.append({
                'number': int(match.group(1)),
                'text': current_question.strip()
            })
    
    existing_numbers = {q['number'] for q in questions}
    if existing_numbers:
        max_num = max(existing_numbers)
        for num in range(1, max_num + 1):
            if num not in existing_numbers:
                questions.append({
                    'number': num,
                    'text': f"{num}. [MANUALLY ADDED QUESTION {num}]"
                })
    
    questions.sort(key=lambda x: x['number'])
    print(f"Parsed {len(questions)} questions")
    return questions

def parse_solutions(file_path):
    solutions = []
    current_solution = ""
    current_number = None
    
    with open(file_path, 'r', encoding='utf-8') as file:
        for line in file:
            line = line.strip()
            
            # Check if this line starts a new solution
            match = re.match(r'^(\d+)[).]|^\((\d+)\)|^(\d+)\.', line)
            if match:
                # Save previous solution if exists
                if current_number is not None and current_solution:
                    solutions.append({
                        'number': current_number,
                        'text': current_solution.strip()
                    })
                
                # Extract the number from the match
                num = match.group(1) or match.group(2) or match.group(3)
                current_number = int(num)
                
                # Remove the number prefix from the line
                current_solution = re.sub(r'^(\d+[).]|\(\d+\)|\d+\.)\s*', '', line)
            else:
                # Continue building the current solution
                if current_solution:
                    current_solution += '\n' + line
                else:
                    current_solution = line
    
    # Add the last solution
    if current_number is not None and current_solution:
        solutions.append({
            'number': current_number,
            'text': current_solution.strip()
        })
    
    # Fill in missing solution numbers 
    existing_numbers = {s['number'] for s in solutions}
    if existing_numbers:
        max_num = max(existing_numbers)
        for num in range(1, max_num + 1):
            if num not in existing_numbers:
                solutions.append({
                    'number': num,
                    'text': f"{num}. [MANUALLY ADDED SOLUTION {num}]"
                })
    
    solutions.sort(key=lambda x: x['number'])
    print(f"Parsed {len(solutions)} solutions")
    return solutions

def process_all_questions(questions, solutions):
    """Process all questions with enhanced sub-question handling (from code #2)"""
    objective_data = []
    descriptive_data = []
    subjective_data = []
    solution_dict = {sol['number']: sol['text'] for sol in solutions}

    for question in questions:
        solution = solution_dict.get(question['number'], '[SOLUTION NOT FOUND]')
        question_text = question['text'].strip()
        
        # Check if it's an MCQ (objective)
        options = re.findall(r'^[a-dA-D][).]\s*.+$', question_text, re.MULTILINE)
        
        if options and len(options) == 4:  
            clean_text = re.sub(r'^[a-dA-D][).]\s*.+$', '', question_text, flags=re.MULTILINE).strip()
            
            objective_data.append({
                'Question Label': f'Q{question["number"]}',
                 'Question Category': '',
                'Cognitive Skills': '',
                'Question Source': '',
                'Question Appears in': 'Pre/Post-Worksheet/Test',
                'Level of Difficulty': '',
                'Question': clean_text,
                'Marks': 1,
                'Answer Type1': 'Words', 
                'Answer Content1': options[0] if len(options) > 0 else '',
                'Correct Answer1': 'No',
                'Answer Weightage1': 0,
                'Answer Type2': 'Words',  
                'Answer Content2': options[1] if len(options) > 1 else '',
                'Correct Answer2': 'No',
                'Answer Weightage2': 0,
                'Answer Type3': 'Words',  
                'Answer Content3': options[2] if len(options) > 2 else '',
                'Correct Answer3': 'No',
                'Answer Weightage3': 0,
                'Answer Type4': 'Words',  
                'Answer Content4': options[3] if len(options) > 3 else '',
                'Correct Answer4': 'No',
                'Answer Weightage4': 0,
                'Answer Explanation': solution
            })
        # Check if it's a fill-in-the-blank question 
        elif re.search(r'_{2,}', question_text):
            blank_num = 0
            while re.search(r'__{2,}', question_text):
                letter = chr(97 + blank_num)
                question_text = re.sub(
                    r'_{2,}',
                    f'$${letter}$$',
                    question_text,
                    count=1
                )
                blank_num += 1

            row = {
                'Question Label': f'Q{question["number"]}',
                'Question Category': 'Fill in the Blanks',
                'Cognitive Skills': '',
                'Question Source': 'Upschool DB',
                'Question Appears in': 'Pre/Post-Worksheet/Test',
                'Level of Difficulty': '',
                'Question': question_text,
                'Marks': 1,
                'answer_explanation': solution
                
            }
            
            # Initialize answer columns (to be filled by GPT later)
            for i in range(1, 11):
                if i <= blank_num:
                    row[f'Answer Type{i}'] = 'Words'
                    row[f'Answer{i}'] = ''  # Placeholder for GPT content
                    row[f'Answer Display{i}'] = 'Yes'
                    row[f'Weightage{i}'] = 1
                    row[f'Placeholder{i}'] = chr(96 + i)
                else:
                    row[f'Answer Type{i}'] = ''
                    row[f'Answer{i}'] = ''
                    row[f'Answer Display{i}'] = ''
                    row[f'Weightage{i}'] = ''
                    row[f'Placeholder{i}'] = ''
            
            subjective_data.append(row)
        
        else:
            # Process as descriptive question with sub-question handling
            main_question = []
            sub_questions = []
            
            lines = question_text.split('\n')
            current_sub = None
            
            for line in lines:
                # Detect sub-questions (a), b), i), ii), etc.
                sub_match = re.match(r'^\s*([a-z]\)|\d+\.\d+|\d+\)|[ivx]+\))\s*(.*)', line.strip())
                if sub_match:
                    if current_sub:
                        sub_questions.append(current_sub)
                    current_sub = {
                        'label': sub_match.group(1),
                        'text': sub_match.group(2),
                        'parts': []
                    }
                elif current_sub:
                    # Detect sub-sub-questions (i), ii), etc.
                    sub_sub_match = re.match(r'^\s*([a-z]\)|\d+\.\d+|\d+\)|[ivx]+\))\s*(.*)', line.strip())
                    if sub_sub_match:
                        current_sub['parts'].append({
                            'label': sub_sub_match.group(1),
                            'text': sub_sub_match.group(2)
                        })
                    else:
                        current_sub['text'] += '\n' + line.strip()
                else:
                    main_question.append(line.strip())
            
            if current_sub:
                sub_questions.append(current_sub)
            
            # Format the question text with sub-questions
            formatted_text = '\n'.join(main_question)
            if sub_questions:
                formatted_text += '\n\nSub-questions:\n'
                for sub in sub_questions:
                    formatted_text += f"\n{sub['label']} {sub['text']}"
                    if sub['parts']:
                        formatted_text += '\n'
                        for part in sub['parts']:
                            formatted_text += f"\n  {part['label']} {part['text']}"
            
            descriptive_data.append({
                'Question Label': f'Q{question["number"]}',
                'Question Category': 'Descriptive',
                'Cognitive Skills': '',
                'Question Source': '',
                'Question Appears in': 'Pre/Post-Worksheet/Test',
                'Level of Difficulty': '',
                'Question': formatted_text.replace('\n', ' ').strip(),
                'Marks': 1 if '[1]' in question_text else 2 if '[2]' in question_text else 1,
                'Display Answer': solution.strip(),
                'Answer Type': '',
                'Answer Weightage': '',
                'Answer Content': '',  
                'Answer Explanation': solution.strip()  
            })
    
    return objective_data, descriptive_data, subjective_data

def extract_correct_answer(explanation):
    if not explanation or not isinstance(explanation, str):
        return None
    
    match = re.search(r'[\(\s]([a-dA-D])[\).\s]|correct.?([a-dA-D])|answer.?([a-dA-D])', 
                     explanation, re.IGNORECASE)
    if match:
        return (match.group(1) or match.group(2) or match.group(3)).lower()
    return None

def mark_correct_answers(objective_df):
    for index, row in objective_df.iterrows():
        correct_answer = extract_correct_answer(row['Answer Explanation'])
        
        for i in range(1, 5):
            answer_content = row[f'Answer Content{i}']
            if pd.isna(answer_content):
                objective_df.at[index, f'Correct Answer{i}'] = 'No'
                objective_df.at[index, f'Answer Weightage{i}'] = 0
            elif correct_answer and answer_content.startswith(f"{correct_answer})"):
                objective_df.at[index, f'Correct Answer{i}'] = 'Yes'
                objective_df.at[index, f'Answer Weightage{i}'] = row['Marks'] if not pd.isna(row['Marks']) else 0
            else:
                objective_df.at[index, f'Correct Answer{i}'] = 'No'
                objective_df.at[index, f'Answer Weightage{i}'] = 0
    
    return objective_df

@app.route('/upload', methods=['POST'])
def upload_file():
    """Handle file upload and processing"""
    if 'questionPaper' not in request.files or 'answerSheet' not in request.files:
        return jsonify({'error': 'Missing files'}), 400

    question_paper = request.files['questionPaper']
    answer_sheet = request.files['answerSheet']

    if question_paper.filename == '' or answer_sheet.filename == '':
        return jsonify({'error': 'No selected file'}), 400

    print("Processing files with Mathpix...")
    question_txt_path = process_with_mathpix(question_paper)
    answer_txt_path = process_with_mathpix(answer_sheet)

    if not question_txt_path or not answer_txt_path:
        return jsonify({'error': 'Failed to process files with Mathpix'}), 500

    process_files_to_excel(question_txt_path, answer_txt_path, intermediate_excel_path)

    final_excel_path = "final_output.xlsx"
    process_excel_file_with_gpt(intermediate_excel_path, final_excel_path)
