def extract_marks_from_question(question_text, fallback_marks):
    """
    Extract marks from patterns like [5], [5M], [ 3 m ] anywhere in the question text.
    If not found, fall back to Excel 'Marks' column.
    """
    # First, trust the marks already present in the sheet, if any
    try:
        if fallback_marks not in (None, "", "nan"):
            m_val = int(float(fallback_marks))
            if m_val > 0:
                return m_val
    except Exception:
        pass

    # Otherwise, scan question text for [...], e.g. [5], [5M], [ 3 m ]
    if question_text:
        q = str(question_text).strip()
        m = re.search(r'\[(\d+)\s*[mM]?\]', q)
        if m:
            try:
                return int(m.group(1))
            except Exception:
                pass

    # Safe fallback
    return 1
