def extract_marks_from_question(question_text, fallback_marks):
    """
    Extract marks from patterns like [5], [5M], [ 3 m ] at the end of the question.
    If not found, fall back to Excel 'Marks' column.
    """
    if not question_text:
        try:
            return int(float(fallback_marks)) if fallback_marks not in (None, "", "nan") else 1
        except Exception:
            return 1

    q = str(question_text).strip()
    m = re.search(r'\[(\d+)\s*[mM]?\]\s*$', q)
    if m:
        return int(m.group(1))

    try:
        return int(float(fallback_marks)) if fallback_marks not in (None, "", "nan") else 1
    except Exception:
        return 1


def clean_question_text(q: str) -> str:
    """
    Remove leading question numbers and trailing marks like '..... [3]' from the question.
    """
    if not q:
        return q
    q = q.strip()
    # Remove leading question numbers like "1.", "39.", "Q1.", "Q.49"
    q = re.sub(r'^\s*(?:Q\.?\s*\d+\.?|\d+\.?)\s*', '', q)
    # Remove trailing "..... [4]" or "[3]" style marks (any dots + bracketed number at end)
    q = re.sub(r'\.*\s*\[\s*\d+\s*[mM]?\s*\]\s*$', '', q)
    return q.strip()


def detect_answer_type_from_text(answer_text: str) -> str:
    """
    Decide Answer Type: Phrases | Equation | Image
    """
    if not answer_text:
        return "Phrases"

    lower = answer_text.lower()

    # Image cues
    image_keywords = [
        "draw a diagram", "labelled diagram", "labeled diagram",
        "sketch", "draw the figure", "diagrammatic", "flowchart"
    ]
    if any(kw in lower for kw in image_keywords):
        return "Image"

    # Equation cues – lots of symbols, <, >, =, ^, fractions, etc.
    symbolic_chars = sum(ch in "+-*/=^<>π√" for ch in answer_text)
    digits = sum(ch.isdigit() for ch in answer_text)
    letters = sum(ch.isalpha() for ch in answer_text)

    if (symbolic_chars + digits) > letters * 1.2:
        return "Equation"

    return "Phrases"

            if 'Objective' in sheet_name:
                required_columns = ['Question Category', 'Cognitive Skills', 'Question Source', 
                                   'Level of Difficulty', 'Marks', 'Answer Type']
            elif 'Subjective' in sheet_name:

                        gpt_response = get_subjective_details(row['Question'])
                        answer_content = extract_gpt_answer_content(gpt_response)

                        blank_count = row['Question'].count('$$') // 2
